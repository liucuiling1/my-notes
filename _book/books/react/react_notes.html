(6)获取上个页面的传的参数

getUrlData() {
  let urlArray = window.location.search.substring(1).split("&");
  let urlData = {};
  for (let i = 0; i < urlArray.length; i++) {
    let pair = urlArray[i].split('=');
    urlData[pair[0]] = decodeURI(pair[1]);
  }
  return urlData;
}
